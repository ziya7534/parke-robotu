function addRoom() {
    const roomInputs = document.getElementById('roomInputs');
    const div = document.createElement('div');
    div.className = 'room-input';
    div.innerHTML = `<input type="number" placeholder="Uzunluk (A)" class="roomLength">
                     <input type="number" placeholder="Genişlik (B)" class="roomWidth">`;
    roomInputs.appendChild(div);
}

function calculate() {
    const method = document.getElementById('method').value;
    let parkArea = 0, skLength = 0;

    const parkPrice = parseFloat(document.getElementById('parkPrice').value) || 0;
    const shiltePrice = parseFloat(document.getElementById('shiltePrice').value) || 0;
    const skPrice = parseFloat(document.getElementById('skPrice').value) || 0;
    const shapPrice = parseFloat(document.getElementById('shapPrice').value) || 0;
    const adhesivePrice = parseFloat(document.getElementById('adhesivePrice').value) || 0;
    const workPrice = parseFloat(document.getElementById('workPrice').value) || 0;

    if(method==='area') {
        parkArea = parseFloat(document.getElementById('parkArea').value) || 0;
        skLength = parseFloat(document.getElementById('skLength').value) || 0;
    } else if(method==='flat') {
        const flatArea = parseFloat(document.getElementById('flatArea').value) || 0;
        parkArea = flatArea;
        skLength = Math.sqrt(flatArea)*4;
    } else if(method==='detailed') {
        const lengths = document.querySelectorAll('.roomLength');
        const widths = document.querySelectorAll('.roomWidth');
        parkArea = 0;
        skLength = 0;
        for(let i=0;i<lengths.length;i++){
            const l=parseFloat(lengths[i].value)||0;
            const w=parseFloat(widths[i].value)||0;
            parkArea += l*w;
            skLength += 2*(l+w);
        }
    }

    // Zaiyat %5
    parkArea *= 1.05;
    skLength *= 1.05;

    const parkCost = parkArea*parkPrice;
    const shilteCost = parkArea*shiltePrice;
    const skCost = skLength*skPrice;
    const shapCost = document.getElementById('smartShap').value==='yes'?shapPrice:0;
    const totalCost = parkCost + shilteCost + skCost + shapCost + adhesivePrice + workPrice;

    document.getElementById('parkQty').innerText = parkArea.toFixed(2)+" m²";
    document.getElementById('skQty').innerText = skLength.toFixed(2)+" mt";
    document.getElementById('parkCost').innerText = parkCost.toFixed(2)+" ₺";
    document.getElementById('shilteCost').innerText = shilteCost.toFixed(2)+" ₺";
    document.getElementById('skCost').innerText = skCost.toFixed(2)+" ₺";
    document.getElementById('shapCost').innerText = shapCost.toFixed(2)+" ₺";
    document.getElementById('adhesiveCost').innerText = adhesivePrice.toFixed(2)+" ₺";
    document.getElementById('workCost').innerText = workPrice.toFixed(2)+" ₺";
    document.getElementById('totalCost').innerText = totalCost.toFixed(2)+" ₺";
}
